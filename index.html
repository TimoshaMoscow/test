📝 Пример кода инициализации

// Создание экземпляра плеера
const player = new TPlayer('#tplayer-container', {
    sources: {
        '360p': 'test.mp4',
        '720p': 'test.mp4',
        '1080p': 'test.mp4'
    },
    poster: 'poster.jpg',  // Постер видео (опционально)
    accentColor: '#ff0000'  // Акцентный цвет
});

// Управление плеером через API
player.togglePlay();    // Воспроизвести/Пауза
player.toggleMute();    // Выключить/Включить звук
player.skip(10);        // Перемотка на 10 секунд
player.changeQuality('720p');  // Смена качества
player.changeSpeed(1.5);       // Смена скорости
player.destroy();       // Уничтожение плеера
